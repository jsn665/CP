# include <bits/stdc++.h>
# define ll long long int
# define pb push_back
using namespace std;


class Solution {
public:
    int maxConsecutiveAnswers(string answerKey, int k) {
            int N = answerKey.length();
            int j = 0;
            int globalLen = 0;
            int origK = k;
            for(int i = 0; i < N; i++){
                if(answerKey[i] == 'F' && k > 0){
                        k--;
                }
                else if(answerKey[i] == 'F' && k == 0){
                        while(j < i && answerKey[j] != 'F'){
                            j++;
                        }
                        j++;
                }
                globalLen = max(globalLen, i - j + 1);
            }
            k = origK;
            j = 0;
            for(int i = 0; i < N; i++){
                if(answerKey[i] == 'T' && k > 0){
                        k--;
                }
                else if (answerKey[i] == 'T' && k == 0){
                        while(j < i && answerKey[j] != 'T'){
                            j++;
                        }
                        j++;
                }
                globalLen = max(globalLen, i - j + 1);
            }
            return globalLen;
    }
};
